    class Solution {
public:
    /*
    f(n) = max{f(n-1),f(n-1)+a[n]}
    f(0) = max{0,a[0]}
    */
    int FindGreatestSumOfSubArray(vector<int> array) {
        int maxSum = 0x80000000;// еҲқе§ӢеҢ–дёәжңҖе°Ҹзҡ„зҡ„иҙҹж•ҙж•°0x80000000пјҢеӯҳеӮЁе…ЁеұҖжңҖеӨ§еӯҗеәҸе’Ң

        int tmpSum = 0;// жҡӮж—¶еӯҳеӮЁйҒҚеҺҶеҲ°з¬¬iдёӘж•°еӯ—зҡ„еӯҗеәҸеҲ—е’Ң
        for (int i = 0;i < array.size();++i){
            if (tmpSum <= 0) {// еҮәзҺ°иҙҹж•°пјҢзӣҙжҺҘжҠӣејғеүҚйқўзҡ„еӯҗеәҸе’Ң
                tmpSum = array[i];
            } else {// еҗҰеҲҷз»§з»ӯеҸ еҠ 
                tmpSum += array[i];
            }

            if (tmpSum > maxSum) {// жӣҙж–°жңҖеӨ§еӯҗеәҸе’Ң
                maxSum = tmpSum;
            }
        }
        return maxSum;
    }


};
